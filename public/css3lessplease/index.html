Animation run

                    

.box-animation (@animation, @timer, @length) {
    -webkit-animation: @animation @timer @length;
    -moz-animation: @animation @timer @length;
    -o-animation: @animation @timer @length;
    animation: @animation @timer @length;
}
@-webkit-keyframes myanim {
    0%   { opacity: 0.0; }
    50%  { opacity: 0.5; }
    100% { opacity: 1.0; }
}
@-moz-keyframes myanim {
    0%   { opacity: 0.0; }
    50%  { opacity: 0.5; }
    100% { opacity: 1.0; }
}
@-o-keyframes myanim {
    0%   { opacity: 0.0; }
    50%  { opacity: 0.5; }
    100% { opacity: 1.0; }
}
@keyframes myanim {
    0%   { opacity: 0.0; }
    50%  { opacity: 0.5; }
    100% { opacity: 1.0; }
}

.box {
    .box-animation(myanim, 5s, infinite);
}
